class LinkedList {
 private Node first, last;
 LinkedList(int v) { first = last = new Node(v); }
 void add(int v) { 
   Node tmp = last; 
   last = new Node(v);
   tmp.next = last;
 }
 void append(LinkedList l) {
   last.next = l.first;
 }
 static class Node {
   int val; Node next;
   Node(int v) { this.val = v; }
